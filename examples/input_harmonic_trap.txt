  L = 30
  Nup = 10
  Ndown = 10
  U = 4.0
  bc = 'open'

  density_tol = 1.0e-6
  mixing_alpha = 0.3
  verbose = .true.

  output_prefix = 'harmonic_trap'
  save_density = .true.
  save_eigenvalues = .true.
