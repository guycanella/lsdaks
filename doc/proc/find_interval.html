    function find_interval(x_grid, n, x) result(i)
        real(dp), intent(in) :: x_grid(0:), x
        integer, intent(in) :: n
        integer :: left, right, mid, i

        if (x <= x_grid(0)) then
            i = 0
            return
        end if

        if (x >= x_grid(n-1)) then
            i = n - 2
            return
        end if

        left = 0
        right = n - 1

        do while (right - left > 1)
            mid = (left + right) / 2
            if (x < x_grid(mid)) then
                right = mid
            else
                left = mid
            end if
        end do

        i = left
    end function find_interval
